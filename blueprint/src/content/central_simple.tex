\section{Simple Rings}\label{sec:simple-ring}

\begin{definition}[Simple Ring] A ring $R$ is simple if the only two-sided-ideals of $R$ are ${0}$ and $R$.
  \leanok
  \lean{IsSimpleRing}
\end{definition}
